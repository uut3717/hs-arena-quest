function onEdit(e) {
  if (!e || !e.source || !e.range) {
    console.error("이벤트 객체가 올바르지 않습니다.");
    return;
  }

  const sheet = e.source.getActiveSheet();
  const targetSheetId = 1497669811; // 시트 ID (gid=1497669811)

  // 현재 시트의 ID가 대상 시트 ID와 다르면 종료
  if (sheet.getSheetId() !== targetSheetId) return;

  const range = e.range;
  const row = range.getRow();
  const col = range.getColumn();

  // F열부터 경험치 입력 감지 (50판까지)
  if (col >= 6 && row > 1) {
    updateExperience(sheet, row);
    applyVisualEffects(sheet, row);
  }
}

function updateExperience(sheet, row) {
  const maxGames = 50; // 최대 판수
  const minGames = 5; // 최소 판수
  const experienceRange = sheet.getRange(row, 6, 1, maxGames); // F열부터 50판까지의 범위
  const experiences = experienceRange.getValues()[0];

  // 숫자만 합산
  const totalExperience = experiences.reduce((sum, exp) => {
    if (typeof exp === "number" && !isNaN(exp)) {
      return sum + exp;
    }
    return sum;
  }, 0);

  const playedGames = experiences.filter(exp => typeof exp === "number" && !isNaN(exp)).length;

  // 총 경험치 업데이트 (C열)
  sheet.getRange(row, 3).setValue(totalExperience);

  // 총 판수 업데이트 (D열)
  sheet.getRange(row, 4).setValue(playedGames);

  // 최소 5판 달성 여부 확인 (E열)
  if (playedGames >= minGames) {
    sheet.getRange(row, 5).setValue("완료");
  } else {
    sheet.getRange(row, 5).setValue(`${playedGames}/${minGames}`);
  }
}

function applyVisualEffects(sheet, row) {
  const totalExperience = sheet.getRange(row, 3).getValue(); // C열 (총 경험치)
  const minExperience = 5000;

  // 5000점을 채운 캐릭터는 이름만 보이도록 처리
  if (totalExperience >= minExperience) {
    // B열 (스파크라인)은 건드리지 않음
    sheet.getRange(row, 1).setFontColor("black"); // 이름만 검은색으로 표시
    sheet.getRange(row, 3, 1, 8).setBackground("white").setFontColor("white"); // C열부터 J열까지 흰색 처리
  } else {
    // 5000점 미만 캐릭터는 시각적 효과 적용
    sheet.getRange(row, 1, 1, 10).setBackground(null).setFontColor("black"); // A열부터 J열까지 초기화
  }
}
